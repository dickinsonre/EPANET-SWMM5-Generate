#### **2. Strict EPANET 2.3 / SWMM 5.2.4 Compliance**
To ensure strict compliance, we must validate the generated`.inp` content against the official specifications.

**Key Compliance Checks:**
1.  **EPANET 2.3:**
    *   **SECTION ORDER:**`[TITLE]`,`[JUNCTIONS]`,`[RESERVOIRS]`,`[TANKS]`,`[PIPES]`,`[PUMPS]`,`[VALVES]`,`[PATTERNS]`,`[CURVES]`,`[CONTROLS]`,`[RULES]`,`[OPTIONS]`,`[COORDINATES]`,`[VERTICES]`,`[LABELS]`,`[BACKDROP]`,`[END]`.
    *   **DATA TYPES:** Ensure coordinates are`X Y` (floats), diameters are integers (inches), and C-factors are integers (Hazen-Williams).
    *   **UNITS:**`GPM` for flow,`FT` for length/diameter (US Customary).

2.  **SWMM 5.2.4:**
    *   **SECTION ORDER:**`[TITLE]`,`[OPTIONS]`,`[EVAPORATION]`,`[SUBCATCHMENTS]`,`[SUBAREAS]`,`[INFILTRATION]`,`[JUNCTIONS]`,`[OUTFALLS]`,`[CONDUITS]`,`[PUMPS]`,`[ORIFICES]`,`[WEIRS]`,`[XSECTIONS]`,`[TRANSECTS]`,`[CONTROLS]`,`[POLLUTANTS]`,`[LANDUSES]`,`[COVERAGES]`,`[BUILDUP]`,`[WASHOFF]`,`[RDII]`,`[HYDROGRAPHS]`,`[TIMESERIES]`,`[PATTERNS]`,`[CURVES]`,`[COORDINATES]`,`[VERTICES]`,`[LABELS]`,`[BACKDROP]`,`[END]`.
    *   **VALIDATION:** Ensure`OUTFALL` nodes have valid elevation and type (FREE, FIXED, Tide).

**Code Implementation (Snippet):**
```javascript
class InpGenerator {
    constructor(config) {
        this.config = config;
        this.lines = [];
    }

    generateEPANET() {
        this.lines.push(`[TITLE]`, `; Generated by EPANET/SWMM Generator ${new Date().toISOString()}`);
        
        // Strict Section Order
        this._addJunctions();
        this._addReservoirs();
        this._addTanks();
        this._addPipes(); // Validates C-factor range (80-150)
        this._addPumps();
        this._addValves();
        this._addPatterns();
        this._addCurves();
        // ... Continue for all sections
        
        // Validate Options
        this.lines.push(`[OPTIONS]`);
        this.lines.push(`FLOW_UNITS ${this.config.units === 'SI' ? 'LPS' : 'GPM'}`);
        this.lines.push(`HEADLOSS ${this.config.headlossMethod}`); // Hazen-Williams, Darcy-Weisbach
        
        return this.lines.join('\n');
    }

    _addPipes() {
        this.lines.push(`[PIPES]`);
        this.lines.push(`; ID Node1 Node2 Length Diameter C-factor MinorLoss Status`);
        this.pipes.forEach(pipe => {
            // Strict Validation: Diameter must be > 0
            if (pipe.diameter <= 0) pipe.diameter = 6; // Default minimum
            // C-factor clamp
            let c = Math.max(80, Math.min(150, pipe.cFactor));
            this.lines.push(`${pipe.id} ${pipe.node1} ${pipe.node2} ${pipe.length.toFixed(2)} ${pipe.diameter} ${c} 0 OPEN`);
        });
    }
}
```

#### **3. Batch Export with Varied Seeds**
This uses a Web Worker to generate multiple`.inp` files without freezing the UI. We generate a ZIP file containing all files.

**Dependencies:** We will include`JSZip` via CDN (or inline minified) for client-side zipping.

**Code Implementation (Snippet):**
```javascript
// Main Thread
async function generateBatch() {
    const seeds = [];
    const count = parseInt(document.getElementById('batchCount').value) || 10;
    
    // Generate seeds based on user input or random
    for(let i=0; i<count; i++) seeds.push(Math.floor(Math.random() * 10000));

    const zip = new JSZip();
    const folder = zip.folder("batch_models");

    // Use Web Worker for parallel generation
    const worker = new Worker(URL.createObjectURL(new Blob([`
        self.onmessage = function(e) {
            const { config, seeds } = e.data;
            const results = [];
            
            seeds.forEach(seed => {
                // Clone config and apply seed
                const localConfig = JSON.parse(JSON.stringify(config));
                localConfig.randomSeed = seed;
                
                // Run generation logic here (simplified)
                const generator = new InpGenerator(localConfig);
                const content = generator.generateEPANET();
                results.push({ seed, content });
            });
            
            self.postMessage(results);
        };
    `])));

    worker.onmessage = function(e) {
        e.data.forEach(model => {
            folder.file(`model_seed_${model.seed}.inp`, model.content);
        });
        
        zip.generateAsync({type:"blob"}).then(function(content) {
            saveAs(content, "batch_sensitivity_analysis.zip");
        });
    };

    worker.postMessage({ config: currentConfig, seeds: seeds });
}
```

#### **4. "Quick Sim" Button & Heatmap Overlay**
This integrates a lightweight hydraulic solver directly in the browser using a simplified WASM module or JavaScript implementation (e.g., EPANET-MSX logic).

**Implementation Strategy:**
1.  **WASM Integration:** Load the EPANET engine compiled to WebAssembly (WASM) via a CDN or local file.
2.  **Simulation:** Pass the generated`.inp` content to the WASM module.
3.  **Visualization:** Parse the output (pressure/heads) and overlay a color gradient on the canvas.

**Code Structure:**
```javascript
// Load WASM Engine (pseudo-code)
async function loadEngine() {
    const engine = await WebAssembly.instantiateStreaming(fetch('epanet.wasm'));
    return engine.instance.exports;
}

// Quick Sim Function
async function runQuickSim() {
    const inpContent = currentGenerator.generateEPANET();
    
    // Write to virtual file system (WASM requirement)
    FS.writeFile('model.inp', inpContent);
    
    // Run solver
    const engine = await loadEngine();
    engine.ENopen('model.inp', 'report.rpt', '');
    engine.ENsolveH();
    
    // Retrieve results
    const nodeCount = engine.ENgetcount(EN_NODECOUNT);
    const pressures = [];
    for(let i=1; i<=nodeCount; i++) {
        const pressure = engine.ENgetnodevalue(i, EN_PRESSURE);
        pressures.push(pressure);
    }
    
    // Overlay Heatmap
    drawHeatmap(pressures);
}

function drawHeatmap(pressures) {
    const ctx = canvas.getContext('2d');
    nodes.forEach((node, index) => {
        const p = pressures[index];
        const color = getPressureColor(p); // Blue (low) to Red (high)
        ctx.fillStyle = color;
        ctx.beginPath();
        ctx.arc(node.x, node.y, 5, 0, Math.PI * 2);
        ctx.fill();
    });
}
```

#### **5. Interactive Canvas with Click-to-Inspect**
We move from static image generation to an interactive HTML5 Canvas.

**Code Implementation (Snippet):**
```javascript
const canvas = document.getElementById('networkCanvas');
const ctx = canvas.getContext('2d');

// Render Loop
function render() {
    ctx.clearRect(0, 0, canvas.width, canvas.height);
    
    // Draw City Map (background)
    drawCityMap();
    
    // Draw Pipes
    ctx.strokeStyle = '#333';
    ctx.lineWidth = 2;
    pipes.forEach(pipe => {
        ctx.beginPath();
        ctx.moveTo(pipe.node1.x, pipe.node1.y);
        ctx.lineTo(pipe.node2.x, pipe.node2.y);
        ctx.stroke();
    });
    
    // Draw Nodes
    nodes.forEach(node => {
        ctx.fillStyle = node.type === 'reservoir' ? 'blue' : 'black';
        ctx.beginPath();
        ctx.arc(node.x, node.y, 4, 0, Math.PI * 2);
        ctx.fill();
    });
}

// Click-to-Inspect
canvas.addEventListener('click', (e) => {
    const rect = canvas.getBoundingClientRect();
    const x = e.clientX - rect.left;
    const y = e.clientY - rect.top;
    
    // Find nearest node (simple distance check)
    const clickedNode = nodes.find(node => 
        Math.sqrt((node.x - x)**2 + (node.y - y)**2) < 10
    );
    
    if (clickedNode) {
        showInspectPanel(clickedNode);
    }
});

function showInspectPanel(node) {
    const panel = document.getElementById('inspectPanel');
    panel.innerHTML = `
        <h3>Node: ${node.id}</h3>
        <p>Type: ${node.type}</p>
        <p>Elevation: ${node.elevation} ft</p>
        <p>Demand: ${node.demand} GPM</p>
        <p>Pressure: ${node.pressure || 'N/A'} psi</p>
    `;
    panel.style.display = 'block';
}
```
