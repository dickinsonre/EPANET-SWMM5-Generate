## 2. Implementing the`InpGenerator` Class

This class handles strict EPANET 2.3 and SWMM 5.2.4 compliance.

### Step 1: Class Structure
```javascript
class InpGenerator {
  constructor(config) {
    this.config = config;
    this.lines = [];
    this.nodes = [];
    this.pipes = [];
  }

  generateEPANET() {
    this.lines = [];
    this._addTitle();
    this._addJunctions();
    this._addReservoirs();
    this._addTanks();
    this._addPipes();
    this._addPumps();
    this._addValves();
    this._addPatterns();
    this._addCurves();
    this._addOptions();
    this._addCoordinates();
    this._addVertices();
    this._addLabels();
    this._addBackdrop();
    this._addEnd();
    return this.lines.join('\n');
  }

  generateSWMM() {
    this.lines = [];
    this._addTitle();
    this._addOptions();
    this._addSubcatchments();
    this._addSubareas();
    this._addInfiltration();
    this._addJunctions();
    this._addOutfalls();
    this._addConduits();
    this._addPumps();
    this._addOrifices();
    this._addWeirs();
    this._addXSections();
    this._addPollutants();
    this._addLandUses();
    this._addCoverages();
    this._addBuildup();
    this._addWashoff();
    this._addRDII();
    this._addHydrographs();
    this._addTimeseries();
    this._addPatterns();
    this._addCurves();
    this._addCoordinates();
    this._addVertices();
    this._addLabels();
    this._addBackdrop();
    this._addEnd();
    return this.lines.join('\n');
  }

  // Private methods for each section
  _addTitle() {
    this.lines.push(`[TITLE]`);
    this.lines.push(`; Generated by EPANET/SWMM Generator ${new Date().toISOString()}`);
    this.lines.push(`; Grid: ${this.config.gridRows}x${this.config.gridCols}, Terrain: ${this.config.terrainType}`);
  }

  _addJunctions() {
    this.lines.push(`[JUNCTIONS]`);
    this.lines.push(`; ID Elev Demand Pattern`);
    this.nodes.forEach(node => {
      if (node.type === 'junction') {
        this.lines.push(`${node.id} ${node.elevation.toFixed(2)} ${node.demand.toFixed(2)} ${node.pattern}`);
      }
    });
  }

  _addReservoirs() {
    this.lines.push(`[RESERVOIRS]`);
    this.lines.push(`; ID Head Pattern`);
    this.nodes.forEach(node => {
      if (node.type === 'reservoir') {
        this.lines.push(`${node.id} ${node.head.toFixed(2)} ${node.pattern}`);
      }
    });
  }

  _addPipes() {
    this.lines.push(`[PIPES]`);
    this.lines.push(`; ID Node1 Node2 Length Diameter C-factor MinorLoss Status`);
    this.pipes.forEach(pipe => {
      // Strict Validation: Diameter must be > 0
      if (pipe.diameter <= 0) pipe.diameter = 6; // Default minimum
      // C-factor clamp (80-150 for EPANET)
      let c = Math.max(80, Math.min(150, pipe.cFactor));
      this.lines.push(`${pipe.id} ${pipe.node1} ${pipe.node2} ${pipe.length.toFixed(2)} ${pipe.diameter} ${c} 0 OPEN`);
    });
  }

  _addOptions() {
    this.lines.push(`[OPTIONS]`);
    this.lines.push(`FLOW_UNITS ${this.config.units === 'SI' ? 'LPS' : 'GPM'}`);
    this.lines.push(`HEADLOSS ${this.config.headlossMethod}`);
    this.lines.push(`VISCOITY ${this.config.viscosity || 1.0}`);
    this.lines.push(`DIFFUSION ${this.config.diffusion || 1.0}`);
    this.lines.push(`SPECIFIC_GRAVITY ${this.config.specificGravity || 1.0}`);
    this.lines.push(`MAX_TRIES ${this.config.maxTries || 10}`);
    this.lines.push(`ACCURACY ${this.config.accuracy || 0.001}`);
    this.lines.push(`TOLERANCE ${this.config.tolerance || 0.01}`);
    this.lines.push(`EMIT_EXPON ${this.config.emitExpon || 0.5}`);
    this.lines.push(`DEMAND_MULTIPLIER ${this.config.demandMultiplier || 1.0}`);
    this.lines.push(`EMITTER_EXPONENT ${this.config.emitterExponent || 0.5}`);
    this.lines.push(`TORQUE ${this.config.torque || 0.0}`);
    this.lines.push(`CHECK_FREQUENCY ${this.config.checkFrequency || 10}`);
    this.lines.push(`MAX_CHECK_FREQ ${this.config.maxCheckFreq || 100}`);
    this.lines.push(`TEMPERATURE ${this.config.temperature || 20.0}`);
  }

  _addEnd() {
    this.lines.push(`[END]`);
  }
}
```

## 4. Plan for Integrating EPANET WASM Module

### Strategy:
1.  **Source:** Use the official EPA EPANET-MSX WASM build or compile the C source to WASM using Emscripten.
2.  **Integration:** Load the WASM module asynchronously and initialize the EPANET engine.
3.  **Simulation Flow:**
    - Write the generated`.inp` content to the WASM virtual file system.
    - Call EPANET functions (`ENopen`,`ENsolveH`,`ENgetnodevalue`).
    - Parse results and overlay on canvas.

### Implementation Steps:

#### Step 1: Load WASM Module
```javascript
// Assuming epanet.wasm is hosted or embedded
async function loadEPANETWASM() {
  const response = await fetch('epanet.wasm');
  const bytes = await response.arrayBuffer();
  const { instance } = await WebAssembly.instantiate(bytes, {
    env: {
      memory: new WebAssembly.Memory({ initial: 256, maximum: 256 }),
      // Add necessary imports for C runtime (e.g., printf, fopen)
    }
  });
  return instance.exports;
}
```

#### Step 2: Initialize EPANET Engine
```javascript
let epanetEngine = null;

async function initializeEPANET() {
  epanetEngine = await loadEPANETWASM();
  
  // Set up virtual file system (if needed)
  if (epanetEngine.FS) {
    epanetEngine.FS.mkdir('/work');
    epanetEngine.FS.chdir('/work');
  }
}
```

#### Step 3: Run Simulation
```javascript
async function runQuickSim() {
  if (!epanetEngine) await initializeEPANET();
  
  const config = getConfigFromUI();
  const generator = new InpGenerator(config);
  const inpContent = generator.generateEPANET();
  
  // Write to virtual file system
  epanetEngine.FS.writeFile('model.inp', inpContent);
  
  // Run EPANET
  epanetEngine.ENopen('model.inp', 'report.rpt', '');
  epanetEngine.ENsolveH();
  
  // Retrieve results
  const nodeCount = epanetEngine.ENgetcount(0); // 0 = EN_NODECOUNT
  const pressures = [];
  for(let i=1; i<=nodeCount; i++) {
    const pressure = epanetEngine.ENgetnodevalue(i, 2); // 2 = EN_PRESSURE
    pressures.push(pressure);
  }
  
  // Overlay heatmap
  drawHeatmap(pressures);
}
```

```javascript
const canvas = document.getElementById('networkCanvas');
const ctx = canvas.getContext('2d');

function renderNetwork(nodes, pipes) {
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  
  // Draw background map (city streets, etc.)
  drawCityMap();
  
  // Draw pipes
  ctx.strokeStyle = '#333';
  ctx.lineWidth = 2;
  pipes.forEach(pipe => {
    const n1 = nodes.find(n => n.id === pipe.node1);
    const n2 = nodes.find(n => n.id === pipe.node2);
    if (n1 && n2) {
      ctx.beginPath();
      ctx.moveTo(n1.x, n1.y);
      ctx.lineTo(n2.x, n2.y);
      ctx.stroke();
    }
  });
  
  // Draw nodes
  nodes.forEach(node => {
    ctx.fillStyle = node.type === 'reservoir' ? 'blue' : 'black';
    ctx.beginPath();
    ctx.arc(node.x, node.y, 4, 0, Math.PI * 2);
    ctx.fill();
  });
}

function drawCityMap() {
  // Draw grid lines, streets, parks (simplified)
  ctx.strokeStyle = '#ddd';
  ctx.lineWidth = 1;
  for(let i=0; i<canvas.width; i+=50) {
    ctx.beginPath();
    ctx.moveTo(i, 0);
    ctx.lineTo(i, canvas.height);
    ctx.stroke();
  }
  for(let i=0; i<canvas.height; i+=50) {
    ctx.beginPath();
    ctx.moveTo(0, i);
    ctx.lineTo(canvas.width, i);
    ctx.stroke();
  }
}
```

### Step 4: Heatmap Overlay
```javascript
function drawHeatmap(pressures) {
  nodes.forEach((node, index) => {
    const p = pressures[index];
    const color = getPressureColor(p); // Blue (low) to Red (high)
    ctx.fillStyle = color;
    ctx.beginPath();
    ctx.arc(node.x, node.y, 6, 0, Math.PI * 2);
    ctx.fill();
  });
}

function getPressureColor(pressure) {
  // Normalize pressure to 0-1 range (assuming 0-100 psi range)
  const normalized = Math.max(0, Math.min(1, pressure / 100));
  const r = Math.floor(255 * normalized);
  const b = Math.floor(255 * (1 - normalized));
  return `rgb(${r}, 0, ${b})`;
}
```
