## Part 1: Classic EPANET 2.2 (The Foundation)

### What It Is

EPANET 2.2 is the official EPA desktop application for modeling water distribution systems. Released in its current form in 2020, it represents a collaboration between EPA and the Open Water Analytics (OWA) community. It is the direct descendant of Lewis Rossman's original EPANET, first created in the 1990s.

```

Language:        C (engine), Delphi/Pascal (GUI)
Platform:        Windows only (XP through 11)
License:         Public Domain (engine), Free (GUI)
Current Version: 2.2.0 (with patches)
Download:        epa.gov/water-research/epanet
Source Code:     github.com/USEPA/EPANET2.2
File Format:     .inp (input), .rpt (report), .out (binary)
Max Model Size:  Unlimited (tested to 500,000+ nodes)
```

### The EPANET Programmer's Toolkit (API)

The real power of EPANET 2.2 is not the GUI but the **C API** (epanet2.h). This is the engine that all commercial products and Python wrappers use:

```c
// Example: Run a steady-state simulation in C

#include "epanet2_2.h"

int main() {
    EN_Project ph;
    
    // Create project and open input file
    EN_createproject(&ph);
    EN_open(ph, "mymodel.inp", "report.rpt", "");
    
    // Run hydraulic simulation
    EN_solveH(ph);
    
    // Get results: pressure at node index 5
    double pressure;
    EN_getnodevalue(ph, 5, EN_PRESSURE, &pressure);
    printf("Pressure at node 5: %.2f psi\n", pressure);
    
    // Get results: flow in link index 3
    double flow;
    EN_getlinkvalue(ph, 3, EN_FLOW, &flow);
    printf("Flow in link 3: %.2f GPM\n", flow);
    
    // Cleanup
    EN_close(ph);
    EN_deleteproject(ph);
    
    return 0;
}
```
```
KEY API FUNCTION GROUPS:
  ┌──────────────────┬──────────────────────────────────┐
  │ Function Group   │ Purpose                          │
  ├──────────────────┼──────────────────────────────────┤
  │ EN_open/close    │ Load/unload model                │
  │ EN_solveH        │ Run complete hydraulic simulation │
  │ EN_solveQ        │ Run complete WQ simulation        │
  │ EN_openH/nextH   │ Step-by-step hydraulic solving   │
  │ EN_openQ/nextQ   │ Step-by-step WQ solving          │

  │ EN_addnode       │ Add new node to model            │
  │ EN_addlink       │ Add new link to model            │
  │ EN_deletenode    │ Remove node from model           │
  │ EN_deletelink    │ Remove link from model           │
  │ EN_report        │ Generate formatted report        │
  └──────────────────┴──────────────────────────────────┘
```

### WNTR: Getting Started

```python
# Install WNTR
# pip install wntr

import wntr

# Load an EPANET model
wn = wntr.network.WaterNetworkModel('Net3.inp')

# Inspect the network
print(f"Junctions: {wn.num_junctions}")
print(f"Reservoirs: {wn.num_reservoirs}")
print(f"Tanks: {wn.num_tanks}")
print(f"Pipes: {wn.num_pipes}")
print(f"Pumps: {wn.num_pumps}")
print(f"Valves: {wn.num_valves}")

# Run hydraulic simulation using EPANET engine
sim = wntr.sim.EpanetSimulator(wn)
results = sim.run_sim()

# Access results
pressure = results.node['pressure']
flow = results.link['flowrate']
velocity = results.link['velocity']

# Get pressure at node '123' at hour 5
p_at_5hr = pressure.loc[5*3600, '123']
print(f"Pressure at node 123 at t=5hr: {p_at_5hr:.1f} m")

# Plot pressure at all nodes at hour 5
pressure_at_5hr = pressure.loc[5*3600, :]
wntr.graphics.plot_network(
    wn, 
    node_attribute=pressure_at_5hr,
    node_size=30,
    title='Pressure at 5 hours'
)
```

```python
# ═══════════════════════════════════════════
# CAPABILITY 1: PIPE BREAK SIMULATION
# ═══════════════════════════════════════════

# Simulate breaking pipe '123' at hour 12
wn.get_link('123').status = wntr.network.LinkStatus.Closed

# Or more realistically, add a leak
wn = wntr.morph.split_pipe(wn, '123', 'leak_node', 'pipe_A', 'pipe_B')
leak_node = wn.get_node('leak_node')
leak_node.add_leak(wn, area=0.001, start_time=12*3600, end_time=24*3600)

sim = wntr.sim.WNTRSimulator(wn)  # Note: WNTRSimulator, not EpanetSimulator
results = sim.run_sim()

# ═══════════════════════════════════════════
# CAPABILITY 2: RESILIENCE METRICS
# ═══════════════════════════════════════════

# Calculate todini resilience index
todini = wntr.metrics.hydraulic.todini_index(
    pressure.loc[5*3600, :],
    wn,
    results.link['flowrate'].loc[5*3600, :],
    Pstar=21.1  # minimum required pressure in meters (30 psi)
)
print(f"Todini Resilience Index: {todini:.3f}")

# Calculate population impacted by low pressure
pop = wntr.metrics.population_impacted(
    pressure, wn, 
    pressure_threshold=21.1  # 30 psi minimum
)
print(f"Population impacted: {pop}")

# ═══════════════════════════════════════════
# CAPABILITY 3: NETWORK GRAPH ANALYSIS
# ═══════════════════════════════════════════

# Convert to NetworkX graph for topological analysis
G = wn.to_graph()

# Find shortest path between two nodes
import networkx as nx
path = nx.shortest_path(G, source='River', target='Lake')
print(f"Shortest path: {path}")

# Find bridges (critical pipes whose failure disconnects network)
bridges = list(nx.bridges(G.to_undirected()))
print(f"Critical pipes (bridges): {bridges[:5]}")

# Calculate betweenness centrality (most important nodes)
centrality = nx.betweenness_centrality(G)
top_5 = sorted(centrality.items(), key=lambda x: x[1], reverse=True)[:5]

# Set up water age simulation
wn.options.quality.parameter = 'AGE'

sim = wntr.sim.EpanetSimulator(wn)
results = sim.run_sim()

age = results.node['quality']  # Water age in seconds

# Find nodes where water age exceeds 48 hours
stagnant_nodes = age.columns[age.max() > 48*3600]

# Plot water age map
age_at_24hr = age.loc[24*3600, :] / 3600  # Convert to hours
wntr.graphics.plot_network(
    wn,
    node_attribute=age_at_24hr,
    node_size=20,
    title='Water Age at 24 hours (hours)',
    node_colorbar_label='Age (hr)'
)

# ═══════════════════════════════════════════
# CAPABILITY 5: FRAGILITY AND EARTHQUAKE
# ═══════════════════════════════════════════

# Define earthquake scenario
epicenter = (32.3, -110.9)  # Latitude, longitude
magnitude = 6.5
depth = 10  # km

# Calculate peak ground acceleration at each pipe
from scipy.spatial.distance import cdist
# (Simplified — WNTR has full fragility curve support)

# Apply damage to pipes based on PGA
fc = wntr.scenario.FragilityCurve()
fc.add_state('leak', 1, {'Default':
```

### What It Is

The [EPANET/SWMM Network Generator](https://epanet-swmm-5-generate.replit.app/) is a browser-based tool that procedurally generates hydraulically valid water distribution and stormwater models of any size — from 50-node teaching examples to 100,000-node city-scale networks.

```

═══════════════════════

               calibration features, water quality
Author:        Robert Dickinson
License:       Free for education and testing
```

### What Makes Generated Models "Hydraulically Valid"

```
The generator doesn't just scatter random nodes and pipes.
It applies real engineering logic:

1. CONNECTIVITY GUARANTEE
   → BFS (breadth-first search) from all reservoirs
   → Every junction confirmed reachable
   → Isolated nodes: 0 (displayed in Stats tab)

2. PUMP SIZING VIA CRITICAL PATH ANALYSIS
   → BFS traces from each reservoir through the network
   → Uses Hazen-Williams friction to find farthest node
   → Pump head = elevation deficit + friction loss + safety margin
   → Three-point curve brackets the operating point
   → Prevents "pump exceeds max flow" warnings

3. PRESSURE ZONE DETECTION
   → If terrain elevation range > 80 ft:
     System auto-divides into zones (~70 ft each)
   → PRVs placed at zone boundary pipes
   → Settings computed: downstream elevation + 50 psi target

4. SUPPLY-DEMAND BALANCE
   → Total supply capacity computed from reservoir heads + pumps
   → Total demand computed from all junction base demands × peak factor
   → Supply/Demand ratio displayed (green = adequate, red = deficit)
   → If S/D < 1.0, model is intentionally undersized for
     calibration practice (learn to diagnose supply problems)

5. REALISTIC SPATIAL PATTERNS
   → Demands not random: Industrial clustered NW, Commercial center
   → C-factors correlated with position: old downtown = low C
   → Pipe diameters grade from transmission to distribution
   → Pipe ages correlated with distance from center (1950s→2020s)
```

## Part 4: Practicing Calibration Techniques

### Exercise 1: Manual Roughness Calibration (Beginner)

```
SETUP:
  1. Generate a "Medium" model (750 nodes) with the generator
  2. Save as "truth_model.inp"
  3. Run in EPANET/WNTR, record pressures at 20 random nodes
     → These are your "field measurements"
  4. Copy file to "calibration_model.inp"
  5. In calibration_model.inp, change ALL C-factors to 120
     (destroy the spatial variation)
  
TASK:
  Adjust C-factors in calibration_model.inp until pressures
  match your "field measurements" within ±3 psi.

RULES:
  → You may NOT look at truth_model.inp during calibration
  → You may only adjust pipe roughness (not demands)
  → Document every change you make and why

PYTHON AUTOMATION:
```
```python
import wntr
import numpy as np

# Load the "truth" model
wn_truth = wntr.network.WaterNetworkModel('truth_model.inp')
sim_truth = wntr.sim.EpanetSimulator(wn_truth)
results_truth = sim_truth.run_sim()

# Select 20 random "monitoring" nodes
np.random.seed(42)
all_junctions = list(wn_truth.junction_name_list)
monitor_nodes = list(np.random.choice(all_junctions, 20, replace=False))

# Record "field measurements" at hour 12 (peak demand)
truth_pressures = {}
for node_id in monitor_nodes:
    p = results_truth.node['pressure'].loc[12*3600, node_id]
    truth_pressures[node_id] = p

print("=== SIMULATED FIELD MEASUREMENTS (hour 12) ===")
for node_id, p in truth_pressures.items():
    print(f"  Node {node_id}: {p:.1f} m ({p*1.422:.1f} psi)")

# Load the "corrupted" calibration model
wn_cal = wntr.network.WaterNetworkModel('calibration_model.inp')

# Set all pipes to C=120 (destroy spatial variation)
for pipe_name, pipe in wn_cal.pipes():
    pipe.roughness = 120

# Run calibration model
sim_cal = wntr.sim.EpanetSimulator(wn_cal)
results_cal = sim_cal.run_sim()

# Compare
print("\n=== CALIBRATION COMPARISON (before adjustment) ===")
print(f"{'Node':<10} {'Truth (m)':<12} {'Model (m)':<12} {'Error (m)':<10}")
print("-" * 45)
total_error = 0
for node_id in monitor_nodes:
    p_truth = truth_pressures[node_id]
    p_model = results_cal.node['pressure'].loc[12*3600, node_id]
    error = p_model - p_truth
    total_error += error**2
    print(f"{node_id:<10} {p_truth:<12.1f} {p_model:<12.1f} {error:<10.1f}")

rmse = np.sqrt(total_error / len(monitor_nodes))
print(f"\nRMSE: {rmse:.2f} m ({rmse*1.422:.2f} psi)")
print(f"Target: < 2.1 m (< 3.0 psi)")
```

### Exercise 2: Automated Calibration with Optimization (Intermediate)

```python
import wntr
import numpy as np
from scipy.optimize import differential_evolution

# Load truth model and get "field data"
wn_truth = wntr.network.WaterNetworkModel('truth_model.inp')
sim = wntr.sim.EpanetSimulator(wn_truth)
results_truth = sim.run_sim()

# Monitor nodes and truth pressures
monitor_nodes = ['J1', 'J50', 'J100', 'J200', 'J350', 
                 'J500', 'J650', 'J750']
truth_p = {n: results_truth.node['pressure'].loc[12*3600, n] 
           for n in monitor_nodes}

# Define pipe groups for calibration (by diameter)
# Group 1: Large pipes (>12")  Group 2: Medium (8-12")  
# Group 3: Small (<8")
def get_pipe_groups(wn):
    groups = {'large': [], 'medium': [], 'small': []}
    for name, pipe in wn.pipes():
        d = pipe.diameter * 1000  # Convert to mm
        if d > 300:
            groups['large'].append(name)
        elif d > 200:
            groups['medium'].append(name)
        else:
            groups['small'].append(name)
    return groups

# Objective function: RMSE between model and "field" pressures
def objective(params):
    c_large, c_medium, c_small = params
    
    wn = wntr.network.WaterNetworkModel('truth_model.inp')
    groups = get_pipe_groups(wn)
    
    # Apply grouped C-factors
    for name in groups['large']:
        wn.get_link(name).roughness = c_large
    for name in groups['medium']:
        wn.get_link(name).roughness = c_medium
    for name in groups['small']:
        wn.get_link(name).roughness = c_small
    
    try:
        sim = wntr.sim.EpanetSimulator(wn)
        results = sim.run_sim()
        
        errors = []
        for node_id in monitor_nodes:
            p_model = results.node['pressure'].loc[12*3600, node_id]
            p_truth = truth_p[node_id]
            errors.append((p_model - p_truth)**2)
        
        return np.sqrt(np.mean(errors))
    except:
        return 1000  # Penalty for failed simulation

# Run differential evolution optimizer
bounds = [(60, 150), (60, 150), (60, 150)]  # C-factor range
result = differential_evolution(
    objective, 
    bounds, 
    maxiter=50, 
    seed=42,
    tol=0.01,
    disp=True
)

print(f"\nOptimal C-factors:")
print(f"  Large pipes:  {result.x[0]:.0f}")
print(f"  Medium pipes: {result.x[1]:.0f}")
print(f"  Small pipes:  {result.x[2]:.0f}")
print(f"  Final RMSE:   {result.fun:.2f} m ({result.fun*1.422:.2f} psi)")
```

### Benchmark 3: Convergence Stress Test

```python
import wntr
import numpy as np

# Test how EPANET handles progressively harder problems

def stress_test_convergence(model_path, demand_multipliers):
    """
    Increase demands until the solver fails to converge.
    This reveals the model's "breaking point."
    """
    results_log = []
    
    for mult in demand_multipliers:
        wn = wntr.network.WaterNetworkModel(model_path)
        
        # Scale all demands
        for junc_name, junc in wn.junctions():
            junc.demand_timeseries_list[0].base_value *= mult
        
        # Set tight convergence
        wn.options.hydraulic.accuracy = 0.001
        wn.options.hydraulic.trials = 40
        wn.options.time.duration = 0  # Steady-state
        
        try:
            sim = wntr.sim.EpanetSimulator(wn)
            results = sim.run_sim()
            
            # Check for negative pressures
            p = results.node['pressure'].loc[0, :]
            neg_count = (p < 0).sum()
            min_p = p.min()
            
            results_log.append({
                'multiplier': mult,
                'converged': True,
                'neg_pressure_nodes': neg_count,
                'min_pressure_m': min_p,
                'mean_pressure_m': p.mean()
            })
            
            status = "✓" if neg_count == 0 else f"⚠ {neg_count} neg"
            print(f"  Demand ×{mult:.1f}: {status}, "
                  f"min P = {min_p:.1f} m ({min_p*1.422:.1f} psi)")
            
        except Exception as e:
            results_log.append({
                'multiplier': mult,
                'converged': False,
                'error': str(e)
            })
            print(f"  Demand ×{mult:.1f}: ✗ FAILED — {str(e)[:60]}")
            break
    
    return results_log

# Run stress test
print("=== CONVERGENCE STRESS TEST ===")
print("Progressively increasing demands until failure:\n")

multipliers = [0.5, 1.0, 1.5, 2.0, 2.5, 3.0, 4.0, 5.0, 7.5, 10.0]
log = stress_test_convergence('medium_model.inp', multipliers)

# Plot results
import matplotlib.pyplot as plt

converged = [r for r in log if r['converged']]
if converged:
    mults = [r['multiplier'] for r in converged]
    min_p = [r['min_pressure_m'] * 1.422 for r in converged]  # psi
    neg_n = [r['neg_pressure_nodes'] for r in converged]
    
    fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(10, 8), sharex=True)
    
    ax1.plot(mults, min_p, 'ro-', linewidth=2)
    ax1.axhline(y=0, color='r', linestyle='--', label='Zero pressure')
    ax1.axhline(y=20, color='orange', linestyle='--', label='Min standard (20 psi)')
    ax1.set_ylabel('Minimum Pressure (psi)')
    ax1.legend()
    ax1.grid(True)
    ax1.set_title('Convergence Stress Test')
    
    ax2.bar(mults, neg_n, color='red', alpha=0.7)
    ax2.set_xlabel('Demand Multiplier')
    ax2.set_ylabel('Nodes with Negative Pressure')
    ax2.grid(True)
    
    plt.tight_layout()
    plt.savefig('stress_test.png', dpi=150)
    plt.show()
```

```python
import wntr

def benchmark_eps(model_path, timesteps_minutes):
    """
    Compare EPS execution time at different hydraulic timesteps.
    Smaller timestep = more accurate but slower.
    """
    print(f"\n{'Timestep':<12} {'# Steps':<10} {'Time (s)':<12} {'Max Tank ΔL':<15}")
    print("=" * 50)
    
    results_dict = {}
    
    for ts_min in timesteps_minutes:
        wn = wntr.network.WaterNetworkModel(model_path)

        wn.options.time.report_timestep = 3600  # Report hourly
        
        start = time.time()
        sim = wntr.sim.EpanetSimulator(wn)
        results = sim.run_sim()
        elapsed = time.time() - start
        
        n_steps = int(24 * 60 / ts_min)
        
        # Check tank level variation
        if wn.num_tanks > 0:
            tank_name = list(wn.tank_name_list)[0]
            tank_levels = results.node['pressure'].loc[:, tank_name]
            max_delta = tank_levels.max() - tank_levels.min()
        else:
            max_delta = 0
        
        results_dict[ts_min] = results
        print(f"{ts_min} min{'':<7} {n_steps:<10} {elapsed:<12.3f} {max_delta:<15.2f}")
    
    return results_dict

# Run benchmark
timesteps = [1, 5, 15, 30, 60]

    p_60min = results[60].node['pressure']

    for t in common_times:
        diff = (p_1min.loc[t, :] - p_60min.loc[t, :]).abs()
```
